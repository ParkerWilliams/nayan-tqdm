```python
"""cat_walk theme -- a cat walking across the progress bar.

Single-line theme using WALK mode. The cat moves left-to-right
as progress increases.
"""
from __future__ import annotations

from ..models import Animation, AnimationMode, Frame
from ..registry import register_theme


def _create(tier: str) -> Animation:
    """Create cat_walk Animation at the given rendering tier."""
    if tier == "emoji":
        frames = _emoji_frames()
        completion = _emoji_completion()
    elif tier == "unicode":
        frames = _unicode_frames()
        completion = _unicode_completion()
    else:
        frames = _ascii_frames()
        completion = _ascii_completion()
    return Animation(
        name="cat_walk",
        frames=frames,
        fps=4.0,
        mode=AnimationMode.WALK,
        completion_frame=completion,
    )


def _emoji_frames() -> tuple[Frame, ...]:
    # Walking cat with alternating paw positions
    # Use 2 frames for a simple walk cycle
    return (
        Frame(lines=("\U0001f431",)),   # cat face
        Frame(lines=("\U0001f408",)),   # cat body (alternate)
    )

def _emoji_completion() -> Frame:
    return Frame(lines=("\U0001f63a",))  # grinning cat

def _unicode_frames() -> tuple[Frame, ...]:
    return (
        Frame(lines=("/\\_/\\",)),
        Frame(lines=("(\\_/)",)),
    )

def _unicode_completion() -> Frame:
    return Frame(lines=("(=^.^=)",))

def _ascii_frames() -> tuple[Frame, ...]:
    return (
        Frame(lines=("=^.^=",)),
        Frame(lines=("=^_^=",)),
    )

def _ascii_completion() -> Frame:
    return Frame(lines=("=^w^=",))


register_theme("cat_walk", _create)
```

**IMPORTANT frame design rules:**
- All frames within a tier MUST have the same `display_width` and `height`
- Pad shorter art with spaces to match the widest frame
- For emoji: be aware that emoji are typically 2 display columns wide (disp_len handles this)
- Test that `disp_len(frame.lines[0])` is consistent across all frames in the animation
