```python
"""Theme registry for nyanbar.

Maps theme names to lazy factory functions that produce Animation objects
at a specific rendering tier (emoji, unicode, ascii).
"""
from __future__ import annotations

import os
import warnings
from typing import Callable

from .models import Animation
from .terminal import ColorTier, TerminalInfo

__all__ = [
    "register_theme", "get_theme", "list_themes",
    "set_theme", "get_default_theme", "resolve_render_tier",
    "register_theme",
]

# Type alias for theme factory: takes tier string, returns Animation
ThemeFactory = Callable[[str], Animation]

# Registry state (module-level)
_themes: dict[str, ThemeFactory] = {}
_cache: dict[tuple[str, str], Animation] = {}  # (name, tier) -> Animation
_default_theme: str = "nyan"


def register_theme(name: str, factory: ThemeFactory) -> None:
    """Register a theme factory.  Overwrites existing entry."""
    _themes[name] = factory
    # Invalidate cache entries for this name
    keys_to_remove = [k for k in _cache if k[0] == name]
    for k in keys_to_remove:
        del _cache[k]


def get_theme(name: str, tier: str = "emoji") -> Animation:
    """Get Animation for theme *name* at rendering *tier*.

    Lazily calls the factory on first access, caches the result.
    Unknown names warn and fall back to default theme.
    """
    if name not in _themes:
        warnings.warn(
            f"Unknown theme '{name}', using default '{_default_theme}'",
            UserWarning,
            stacklevel=2,
        )
        name = _default_theme
    key = (name, tier)
    if key not in _cache:
        _cache[key] = _themes[name](tier)
    return _cache[key]


def list_themes() -> list[str]:
    """Return sorted list of all registered theme names."""
    return sorted(_themes.keys())


def set_theme(name: str) -> None:
    """Set the global default theme for future bars."""
    global _default_theme
    if name not in _themes:
        warnings.warn(
            f"Unknown theme '{name}', keeping default '{_default_theme}'",
            UserWarning,
            stacklevel=2,
        )
        return
    _default_theme = name


def get_default_theme() -> str:
    """Return the current default theme name."""
    return _default_theme


def _clear_cache() -> None:
    """Clear the theme cache (for testing)."""
    _cache.clear()


def resolve_render_tier(
    info: TerminalInfo,
    render_level: str | None = None,
) -> str:
    """Determine the rendering tier for the given terminal.

    Parameters
    ----------
    info:
        Terminal capabilities snapshot.
    render_level:
        User override: ``"emoji"``, ``"unicode"``, or ``"ascii"``.
        If provided, bypasses auto-detection.

    Returns
    -------
    str
        One of ``"emoji"``, ``"unicode"``, ``"ascii"``.
    """
    if render_level is not None:
        return render_level

    # Check UTF-8 locale support
    has_utf8 = _check_utf8_locale()

    if not has_utf8:
        return "ascii"

    if info.color_support >= ColorTier.COLOR_256:
        return "emoji"
    elif info.color_support >= ColorTier.COLOR_16:
        return "unicode"
    else:
        return "ascii"


def _check_utf8_locale() -> bool:
    """Check if the locale supports UTF-8 encoding."""
    for var in ("LC_ALL", "LC_CTYPE", "LANG"):
        val = os.environ.get(var, "")
        if "utf-8" in val.lower() or "utf8" in val.lower():
            return True
    return False
```

New logic: Remove the `ColorTier.NONE` check. TTY terminals with no color support will get ASCII-tier animation (handled by render tier detection in registry.py). Only non-TTY and narrow terminals force fallback.

```python
def should_use_fallback(
    info: TerminalInfo,
    disable: bool = False,
    position: int | None = None,
) -> bool:
    if disable:
        return True
    if not info.is_tty:
        return True
    # Removed: ColorTier.NONE check -- dumb TTY terminals get ASCII-tier animation
    if info.width < 30:
        return True
    if position is not None and position > 0:
        return True
    return False
```
